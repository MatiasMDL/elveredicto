<script>

const SUPABASE_URL="https://ytupmzjhhulcwfuidkzc.supabase.co";

const SUPABASE_KEY="sb_publishable_rKhLyK7Ir9hrh_1aQZxrcg_Sjn-E15g";

const supabaseClient=supabase.createClient(SUPABASE_URL,SUPABASE_KEY);

const nicknameInput=document.getElementById("nickname");

const categoriesContainer=document.getElementById("categoriesContainer");

const questionsCard=document.getElementById("questionsCard");

const questionsList=document.getElementById("questionsList");

const shareCard=document.getElementById("shareCard");

let selectedCategory=null;

let currentQuestions=[];

async function loadCategories(){

const {data,error}=await supabaseClient

.from("categorias")

.select("*")

.eq("activa",true);

const icons={

amigos:"😎",

familia:"👨‍👩‍👧",

pareja:"💘",

picante:"🔥"

};

data.forEach(cat=>{

const btn=document.createElement("div");

btn.className=`ev-category ${cat.slug}`;

btn.innerHTML=`${icons[cat.slug]||"⭐"} ${cat.nombre}`;

btn.onclick=()=>{

document.querySelectorAll(".ev-category")

.forEach(el=>el.classList.remove("active"));

btn.classList.add("active");

selectedCategory=cat.id;

};

categoriesContainer.appendChild(btn);

});

}

async function fetchRandomQuestions(){

const {data}=await supabaseClient

.from("preguntas")

.select("*")

.eq("categoria_id",selectedCategory)

.eq("activa",true);

if(!data||data.length<5){

document.getElementById("errorBox").style.display="block";

document.getElementById("errorBox").innerText="Esta categoría todavía no tiene suficientes preguntas.";

return;

}

currentQuestions=[...data].sort(()=>Math.random()-0.5).slice(0,5);

questionsList.innerHTML="";

currentQuestions.forEach((q,i)=>{

const div=document.createElement("div");

div.className="ev-question";

div.innerHTML=`${i+1}. ${q.texto}`;

questionsList.appendChild(div);

});

questionsCard.style.display="block";

}

document

.getElementById("loadQuestionsBtn")

.addEventListener("click",fetchRandomQuestions);

document

.getElementById("refreshQuestionsBtn")

.addEventListener("click",fetchRandomQuestions);

async function createVerdict(){

const nickname=nicknameInput.value;

const code=Math.random().toString(36).substring(2,8).toUpperCase();

const {data}=await supabaseClient

.from("veredictos")

.insert([{

nickname:nickname,

code:code

}])

.select()

.single();

const rows=currentQuestions.map((q,i)=>({

veredicto_id:data.id,

pregunta_id:q.id,

orden:i+1

}));

await supabaseClient

.from("veredicto_preguntas")

.insert(rows);

const link=window.location.origin+"/v/"+code;

document.getElementById("linkInput").value=link;

shareCard.style.display="block";

}

document

.getElementById("confirmVerdictBtn")

.addEventListener("click",createVerdict);

loadCategories();

</script>
